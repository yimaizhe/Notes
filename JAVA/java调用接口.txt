import java.net.*;
import java.io.*;

public class URLReader {
    public static void main(String[] args) throws Exception {
	URL yahoo = new URL("http://www.baidu.com/query.jsp?param1=value2&param2=value2");
	BufferedReader in = new BufferedReader(
				new InputStreamReader(
				yahoo.openStream()));

	String inputLine;

	while ((inputLine = in.readLine()) != null)
	    System.out.println(inputLine);

	in.close();
    }
}

			String serviceId,String serviceName,String status,String appName, String f) {
		WebTarget res = this.getWebTarget().path("service").path("findByPage")
		.queryParam("pageSize", String.valueOf( pageSize ) )
		.queryParam("pageNo", String.valueOf( pageNo ) )
		.queryParam("serviceId", serviceId )
		.queryParam("serviceName", serviceName )
		.queryParam("status", status )
		.queryParam("appName", appName )
		.queryParam("userId", String.valueOf( userId ) )
		.queryParam("deployable", f);
		String jsonStr = this.addAuthHeader(res.request(), new Date(), ConfigHelper.getValue("restUser")
				, ConfigHelper.getValue("restPwd"), String.valueOf( pageSize ), 
				String.valueOf( pageNo ), String.valueOf( userId ), 
				serviceId,serviceName,status,appName, f  ).get(String.class);
		if( LOG.isDebugEnabled() ){
			LOG.debug("receive message：" + jsonStr);
		}
